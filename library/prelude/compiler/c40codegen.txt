            (\literal literal)
            (\value strFromN value)
            (\value (listAddBack (ListCons 34 value) 34))
            (\o1 \o2 (strConss [nativeExprToExe o1, " ", nativeExprToExe o2, "."]))
nativeDefToExe = \n strCons (n (\literal \o (strConss [literal, " ", nativeExprToExe o]))) "."
